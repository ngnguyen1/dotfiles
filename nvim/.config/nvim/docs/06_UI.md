# Spec: UI Modules

> Hand-rolled statusline and dashboard — zero plugin dependencies.
> Inspired by NvChad's ui plugin but implemented from scratch.

---

## Overview

| Module              | Mounts via                        | Config in                         |
| ------------------- | --------------------------------- | --------------------------------- |
| `ui/statusline.lua` | `vim.opt.statusline = "%!..."`    | active theme's `M.statusline`     |
| `ui/dashboard.lua`  | `require("ui.dashboard").open()`  | `core/config.lua` → `M.dashboard` |
| `ui/cheatsheet.lua` | `require("ui.cheatsheet").open()` | reads keymaps from all buffers    |

## `lua/ui/statusline.lua`

```lua
-- lua/ui/statusline.lua
--
-- Layout matches this reference screenshot:
--
-- LEFT:  [nvim-icon MODE] [file-icon filename branch +added ■modified]
-- RIGHT: [encoding] [✘errors ▲warns] [lsp-icon lsp-name] [file-icon filetype] [≡ line/col]
--
-- Design rules:
--   • Statusline background is TRANSPARENT (bg = "NONE") — items float on editor bg
--   • No powerline arrow/round separators — flat layout, items separated by spacing
--   • Every item has a colored icon preceding its text
--   • Mode pill: colored bg pill for the mode label only; rest of line is transparent
--   • Colors stay in sync with the active theme via highlight groups

local M   = {}
local api = vim.api
local fn  = vim.fn

-- ── MODE DEFINITIONS ─────────────────────────────────────────────
-- icon: Neovim logo before the mode label (matches screenshot 󰣩)
local mode_map = {
  n        = { label = "NORMAL",   hl = "St_NormalMode"  },
  no       = { label = "N·OP",     hl = "St_NormalMode"  },
  v        = { label = "VISUAL",   hl = "St_VisualMode"  },
  V        = { label = "V·LINE",   hl = "St_VisualMode"  },
  ["\22"]  = { label = "V·BLOCK",  hl = "St_VisualMode"  },
  s        = { label = "SELECT",   hl = "St_VisualMode"  },
  S        = { label = "S·LINE",   hl = "St_VisualMode"  },
  i        = { label = "INSERT",   hl = "St_InsertMode"  },
  ic       = { label = "INSERT",   hl = "St_InsertMode"  },
  R        = { label = "REPLACE",  hl = "St_ReplaceMode" },
  Rc       = { label = "REPLACE",  hl = "St_ReplaceMode" },
  c        = { label = "COMMAND",  hl = "St_CommandMode" },
  t        = { label = "TERMINAL", hl = "St_TerminalMode"},
  ["!"]    = { label = "SHELL",    hl = "St_NormalMode"  },
}

-- ── HIGHLIGHT HELPER ─────────────────────────────────────────────
local function h(group, str)
  return "%#" .. group .. "#" .. str
end

-- Spacer on the transparent base
local function gap(n)
  return h("St_Base", string.rep(" ", n or 1))
end

-- ── MODULE: MODE ──────────────────────────────────────────────────
-- Renders: 󰣩 NORMAL
-- The Neovim icon is colored with the mode color; label inside a colored pill
local function module_mode()
  local raw  = api.nvim_get_mode().mode
  local m    = mode_map[raw] or mode_map.n
  -- Icon sits outside the pill, colored same as pill bg → feels "attached"
  return h(m.hl .. "_Icon", " 󰣩 ")
      .. h(m.hl, " " .. m.label .. " ")
      .. gap(1)
end

-- ── MODULE: FILE ──────────────────────────────────────────────────
-- Renders: 󰊢 init.lua
local function module_file()
  local name = fn.expand("%:t")
  if name == "" then name = "[No Name]" end

  local icon     = "󰈚"
  local icon_hl  = "St_FileIcon"
  local ok, dv   = pcall(require, "nvim-web-devicons")
  if ok then
    local i, hl_name = dv.get_icon(name)
    if i then
      icon    = i
      icon_hl = hl_name or icon_hl
    end
  end

  local modified = vim.bo.modified     and h("St_Modified",  " ●") or ""
  local readonly  = vim.bo.readonly    and h("St_Readonly",  " ") or ""

  return h(icon_hl, " " .. icon .. " ")
      .. h("St_FileName", name)
      .. modified
      .. readonly
      .. gap(1)
end

-- ── MODULE: GIT ───────────────────────────────────────────────────
-- Renders:  v2.0  +2  ■3
-- Branch gets  icon, added gets , modified gets ■
local function module_git()
  local gs = vim.b.gitsigns_status_dict
  if not gs or not gs.head or gs.head == "" then return "" end

  local out = {}
  table.insert(out, h("St_GitBranch", " " .. gs.head .. " "))

  if (gs.added   or 0) > 0 then
    table.insert(out, h("St_GitAdded",   " +" .. gs.added .. " "))
  end
  if (gs.changed or 0) > 0 then
    table.insert(out, h("St_GitChanged", "  ~" .. gs.changed .. " "))
  end
  if (gs.removed or 0) > 0 then
    table.insert(out, h("St_GitRemoved", " ■" .. gs.removed .. " "))
  end

  return table.concat(out) .. gap(1)
end

-- ── MODULE: DIAGNOSTICS ───────────────────────────────────────────
-- Renders: ✘ 3  ▲ 2
-- Only shows counts that are > 0
local function module_diagnostics()
  local d   = vim.diagnostic
  local out = {}

  local items = {
    { d.severity.ERROR, "✘", "St_DiagError" },
    { d.severity.WARN,  "▲", "St_DiagWarn"  },
    { d.severity.HINT,  "󰠠", "St_DiagHint"  },
    { d.severity.INFO,  "",  "St_DiagInfo"  },
  }
  for _, item in ipairs(items) do
    local n = #d.get(0, { severity = item[1] })
    if n > 0 then
      table.insert(out, h(item[3], " " .. item[2] .. " " .. n))
    end
  end

  if #out == 0 then return "" end
  return table.concat(out) .. gap(1)
end

-- ── MODULE: ENCODING ──────────────────────────────────────────────
-- Renders: UTF-8
local function module_encoding()
  local enc = vim.bo.fileencoding
  if enc == "" then enc = vim.o.encoding end
  if enc == "" then return "" end
  return h("St_Encoding", enc:upper() .. " ") .. gap(1)
end

-- ── MODULE: LSP ───────────────────────────────────────────────────
-- Renders: 󰣩 null-ls
-- Shows icon + first active LSP client name
local function module_lsp()
  local clients = vim.lsp.get_clients({ bufnr = 0 })
  if #clients == 0 then return "" end

  -- Prefer non-null-ls client if multiple are attached
  local name = clients[1].name
  for _, c in ipairs(clients) do
    if c.name ~= "null-ls" and c.name ~= "copilot" then
      name = c.name
      break
    end
  end

  return h("St_LspIcon", " 󰣩 ") .. h("St_LspName", name .. " ") .. gap(1)
end

-- ── MODULE: FILETYPE ──────────────────────────────────────────────
-- Renders:  nvim   (icon colored by devicons + filetype name)
local function module_filetype()
  local ft = vim.bo.filetype
  if ft == "" then return "" end

  local icon    = ""
  local icon_hl = "St_FtIcon"
  local ok, dv  = pcall(require, "nvim-web-devicons")
  if ok then
    local i, hl_name = dv.get_icon_by_filetype(ft)
    if i then
      icon    = i
      icon_hl = hl_name or icon_hl
    end
  end

  return h(icon_hl, " " .. icon .. " ") .. h("St_FtName", ft .. " ") .. gap(1)
end

-- ── MODULE: PROGRESS ──────────────────────────────────────────────
-- Renders:  40%   (percentage through file, or Top/Bot at edges)
-- Sits between filetype and position in the right segment.
local function module_progress()
  local cur   = fn.line(".")
  local total = fn.line("$")
  local pct
  if cur == 1 then
    pct = "Top"
  elseif cur == total then
    pct = "Bot"
  else
    pct = math.floor(cur / total * 100) .. "%%"
  end
  return h("St_ProgressIcon", "  ") .. h("St_Progress", pct .. " ")
end

-- ── MODULE: POSITION ──────────────────────────────────────────────
-- Renders: ≡ 18/1   (line:col — note colon separator per vim %l:%L syntax)
local function module_position()
  local line = fn.line(".")
  local col  = fn.col(".")
  return h("St_PosIcon", " ≡ ") .. h("St_Pos", line .. ":" .. col .. " ")
end

-- ── SPECIAL WINDOW STATUSLINES ───────────────────────────────────
-- Filetypes that get a minimal, context-specific bar instead of the
-- full editor statusline. Each returns a plain string.

local special_ft = {
  -- NvimTree: show just the directory path, centered
  NvimTree = function()
    local ok, view = pcall(require, "nvim-tree.view")
    local cwd = vim.fn.getcwd()
    cwd = cwd:gsub(vim.env.HOME or "", "~")
    return h("St_NvimTreeSt", "  " .. cwd .. " ")
  end,

  -- Quickfix
  qf = function()
    local title = vim.fn.getqflist({ title = true }).title or "Quickfix"
    return h("St_SpecialSt", "  " .. title .. " ")
  end,

  -- Help
  help = function()
    local name = vim.fn.expand("%:t:r")
    return h("St_SpecialSt", "  " .. name .. " ")
  end,

  -- Lazy / Mason / etc. — just show the filetype capitalized
  lazy   = function() return h("St_SpecialSt", "  lazy.nvim ") end,
  mason  = function() return h("St_SpecialSt", "  Mason ") end,
  lspinfo = function() return h("St_SpecialSt", "  LSP Info ") end,
  checkhealth = function() return h("St_SpecialSt", "  Health ") end,
}

-- ── RENDER ───────────────────────────────────────────────────────
function M.render()
  local ft      = vim.bo.filetype
  local buftype = vim.bo.buftype

  -- 1. Delegate to a special renderer if filetype matches
  if special_ft[ft] then
    return special_ft[ft]()
  end

  -- 2. For any other non-file buffer (terminal, prompt, nofile) show nothing
  if buftype ~= "" and buftype ~= "acwrite" then
    return h("St_Base", "")
  end

  -- 3. Full statusline for normal file buffers
  local base = h("St_Base", "")

  local left = table.concat({
    module_mode(),
    module_file(),
    module_git(),
    module_diagnostics(),
  })

  local right = table.concat({
    module_encoding(),
    module_diagnostics(),
    module_lsp(),
    module_filetype(),
    module_progress(),
    module_position(),
  })

  return base .. left .. "%=" .. right
end

-- ── SETUP ────────────────────────────────────────────────────────
function M.setup()
  -- Global statusline — our render() gates per window via filetype/buftype
  vim.opt.statusline = "%!v:lua.require('ui.statusline').render()"

  -- laststatus=3 → ONE global statusline at the bottom (Neovim 0.8+)
  vim.opt.laststatus = 3

  -- NvimTree sets vim.wo.statusline = "" on its own window to hide the bar.
  -- We override this so our global statusline stays visible even when
  -- NvimTree is focused. The render() function returns the NvimTree-specific
  -- content (CWD path) when ft == "NvimTree".
  local group = vim.api.nvim_create_augroup("StatuslineNvimTree", { clear = true })

  vim.api.nvim_create_autocmd("FileType", {
    group   = group,
    pattern = "NvimTree",
    callback = function(ev)
      -- Force our global expression back on this window.
      -- Must be deferred one tick because NvimTree sets "" after FileType fires.
      vim.schedule(function()
        if vim.api.nvim_buf_is_valid(ev.buf) then
          local win = vim.fn.bufwinid(ev.buf)
          if win ~= -1 then
            vim.wo[win].statusline = "%!v:lua.require('ui.statusline').render()"
          end
        end
      end)
    end,
    desc = "Keep global statusline visible when NvimTree is focused",
  })

  -- Redraw on all relevant events
  vim.api.nvim_create_autocmd(
    { "ModeChanged", "BufEnter", "BufWritePost", "DiagnosticChanged", "LspAttach", "LspDetach" },
    { callback = function() vim.cmd.redrawstatus(true) end }
  )
end

return M
```

## Separator Styles — No Longer Used

The `arrow` / `round` / `block` separator system is **removed** in this layout.
`M.statusline.separator` in theme files can be deleted or left as `"none"`.
The visual separation between items is achieved purely through spacing (`gap()`)
and contrasting icon/text colors against the transparent background.

## `lua/ui/dashboard.lua`

Current behavior:

- Startup header is dynamic: random quote rendered with `cowsay` each launch.
- If the `cowsay` binary is missing, a built-in Lua cowsay fallback is used.
- Header width and action-panel width are normalized to the same centered panel width.
- Action rows keep original commands but are rendered in a padded background card.
- Action key labels (`[ff]`, `[fo]`, ...) are right-aligned to one shared column.
- Footer shows plugin progress (`loaded/total plugins loaded`) + Neovim version.
- Dashboard window is display-only (`nofile`, `readonly`) and disables line numbers.
- Cursor always starts on the first action item and is re-focused on `BufEnter`/`WinEnter`.

Dashboard highlight groups:

  hl(0, "DashboardFooter", { fg = p.grey })
end

## Per-Window Statusline Logic

### The root cause of the duplicate bar

By default Neovim uses `laststatus = 2` — each window gets its own statusline.
With `vim.opt.statusline = "%!expr"`, the expression is re-evaluated for whichever
window Neovim is drawing, so nvim-tree, quickfix, and every split all render the
full bar.

### The fix: `laststatus = 3`

```lua
vim.opt.laststatus = 3   -- single global statusline at the very bottom
```

With `laststatus = 3` (Neovim 0.8+) there is **one statusline** for the entire
editor, not one per window. The `%!expr` is still evaluated per-focused-window,
which means `render()` can still inspect `vim.bo.filetype` to return different
content — but there is only ever one bar visible.

```
┌─────────────────────┬──────────────────────────────────────────┐
│  NvimTree           │  lua/ui/statusline.lua                   │
│                     │                                          │
│  (no statusline     │  (no statusline here either)             │
│   line between      │                                          │
│   these windows)    │                                          │
├─────────────────────┴──────────────────────────────────────────┤
│  NORMAL   statusline.lua   main  +2     LSP ~ lua_ls  lua  98% ≡ 42:7 │
└────────────────────────────────────────────────────────────────┘
```

### Per-window content via `special_ft` table

Even though there is one bar, `render()` still adapts its content based on the
**focused** window's context:

| Focused window    | What `render()` returns                        |
| ----------------- | ---------------------------------------------- |
| Normal file       | Full bar: mode · file · git · diag · lsp · pos |
| NvimTree          | `St_NvimTreeSt`: ` ~/projects/myapp`           |
| Quickfix          | `St_SpecialSt`: ` Quickfix`                    |
| Help              | `St_SpecialSt`: ` helpfile-name`               |
| lazy / mason      | `St_SpecialSt`: ` lazy.nvim` / ` Mason`        |
| Terminal / nofile | Empty (`St_Base` only)                         |

To add a new special window, add an entry to `special_ft`:

```lua
special_ft["yourfiletype"] = function()
  return h("St_SpecialSt", "  Your Label ")
end
```

### `vim.opt.laststatus` must be set in `setup()`, not `options.lua`

`laststatus = 3` only makes sense once the statusline module is active.
Setting it in `options.lua` before plugins load can cause a flash of the
default statusline. Always set it inside `ui/statusline.lua`'s `setup()`.

---

- **Never** `require()` a plugin inside `ui/statusline.lua` at module load time —
  only inside the `render()` function (plugins might not be loaded yet)
- Use `pcall(require, ...)` for all optional plugin dependencies (devicons, gitsigns)
- Dashboard buffer must have `buftype = "nofile"` and `modifiable = false` after writing lines
- All UI modules call `require("core.highlights").load_integration(name)` in their setup,
  not at module load time
- `vim.opt.showtabline = 0` is set in `core/options.lua` to permanently hide the tab bar
